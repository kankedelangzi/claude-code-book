## 📝 实战案例一：Excel 数据清洗

### 需求描述

```
帮我写一个 Python 脚本处理 Excel 数据：

输入：sales.xlsx，包含以下列
- 日期（格式不统一：2024/1/1、2024-01-01、1月1日 混合）
- 商品名称（有些有空格、有些有错别字）
- 金额（有些是文本格式，有些有空格）
- 地区（有些是空值）

处理需求：
1. 统一日期格式为 YYYY-MM-DD
2. 去除商品名称前后空格，统一常见错别字
3. 金额转为数字，去除空格
4. 空值地区标记为"未知"
5. 输出到 sales_cleaned.xlsx
```

### Claude Code 生成的脚本

```python
import pandas as pd
import re
from datetime import datetime

def clean_sales_data(input_file, output_file):
    # 读取 Excel
    df = pd.read_excel(input_file)
    
    # 1. 统一日期格式
    def parse_date(date_str):
        if pd.isna(date_str):
            return None
        
        date_str = str(date_str).strip()
        
        # 尝试多种格式
        formats = [
            '%Y/%m/%d',
            '%Y-%m-%d',
            '%Y年%m月%d日',
            '%m月%d日',  # 补全年份
        ]
        
        for fmt in formats:
            try:
                dt = datetime.strptime(date_str, fmt)
                if fmt == '%m月%d日':
                    dt = dt.replace(year=2024)  # 默认年份
                return dt.strftime('%Y-%m-%d')
            except ValueError:
                continue
        
        return None
    
    df['日期'] = df['日期'].apply(parse_date)
    
    # 2. 清理商品名称
    # 错别字映射
    name_mapping = {
        '苹果手机': 'iPhone',
        '苹 果': 'iPhone',
        'iPhon': 'iPhone',
        '华为手机': 'Huawei',
        '华为手': 'Huawei',
    }
    
    def clean_name(name):
        if pd.isna(name):
            return '未知商品'
        name = str(name).strip()
        return name_mapping.get(name, name)
    
    df['商品名称'] = df['商品名称'].apply(clean_name)
    
    # 3. 清理金额
    def clean_amount(amount):
        if pd.isna(amount):
            return 0
        # 移除空格、逗号等
        amount_str = str(amount).replace(' ', '').replace(',', '').replace('￥', '')
        try:
            return float(amount_str)
        except ValueError:
            return 0
    
    df['金额'] = df['金额'].apply(clean_amount)
    
    # 4. 处理空值地区
    df['地区'] = df['地区'].fillna('未知')
    
    # 5. 输出
    df.to_excel(output_file, index=False)
    print(f"处理完成，共 {len(df)} 条记录")
    return df

# 使用
if __name__ == '__main__':
    clean_sales_data('sales.xlsx', 'sales_cleaned.xlsx')
```

## 📝 实战案例二：日志数据分析

### 需求描述

```
帮我分析 Nginx 访问日志，生成统计报告：

日志格式示例：
192.168.1.1 - - [10/Jan/2024:13:55:36 +0000] "GET /api/users HTTP/1.1" 200 1234

需要统计：
1. 总请求数
2. 各状态码占比
3. Top 10 访问路径
4. Top 10 IP
5. 每小时请求量分布
6. 输出 Markdown 报告
```

### Claude Code 生成的脚本

```python
import re
from collections import Counter
from datetime import datetime

def parse_nginx_log(log_file):
    """解析 Nginx 日志"""
    pattern = r'(\S+) - - \[(.*?)\] "(\S+) (\S+) .*?" (\d+) (\d+)'
    
    records = []
    with open(log_file, 'r') as f:
        for line in f:
            match = re.match(pattern, line)
            if match:
                ip, time_str, method, path, status, size = match.groups()
                dt = datetime.strptime(time_str, '%d/%b/%Y:%H:%M:%S %z')
                records.append({
                    'ip': ip,
                    'datetime': dt,
                    'method': method,
                    'path': path,
                    'status': int(status),
                    'size': int(size)
                })
    
    return records

def analyze_logs(records):
    """分析日志数据"""
    total = len(records)
    
    # 状态码统计
    status_counter = Counter(r['status'] for r in records)
    
    # 路径统计
    path_counter = Counter(r['path'] for r in records)
    
    # IP 统计
    ip_counter = Counter(r['ip'] for r in records)
    
    # 每小时请求量
    hour_counter = Counter(r['datetime'].hour for r in records)
    
    return {
        'total': total,
        'status': status_counter,
        'paths': path_counter.most_common(10),
        'ips': ip_counter.most_common(10),
        'hours': sorted(hour_counter.items())
    }

def generate_report(stats, output_file):
    """生成 Markdown 报告"""
    with open(output_file, 'w') as f:
        f.write('# Nginx 访问日志分析报告\n\n')
        f.write(f'**总请求数**: {stats["total"]}\n\n')
        
        # 状态码分布
        f.write('## 状态码分布\n\n')
        f.write('| 状态码 | 数量 | 占比 |\n')
        f.write('|--------|------|------|\n')
        for status, count in sorted(stats['status'].items()):
            pct = count / stats['total'] * 100
            f.write(f'| {status} | {count} | {pct:.1f}% |\n')
        
        # Top 10 路径
        f.write('\n## Top 10 访问路径\n\n')
        f.write('| 路径 | 访问次数 |\n')
        f.write('|------|----------|\n')
        for path, count in stats['paths']:
            f.write(f'| `{path}` | {count} |\n')
        
        # Top 10 IP
        f.write('\n## Top 10 访问 IP\n\n')
        f.write('| IP | 访问次数 |\n')
        f.write('|----|----------|\n')
        for ip, count in stats['ips']:
            f.write(f'| {ip} | {count} |\n')
        
        # 每小时分布
        f.write('\n## 每小时请求量\n\n')
        f.write('```\n')
        for hour, count in stats['hours']:
            bar = '█' * (count // 100)
            f.write(f'{hour:02d}:00 {bar} {count}\n')
        f.write('```\n')

# 使用
if __name__ == '__main__':
    records = parse_nginx_log('access.log')
    stats = analyze_logs(records)
    generate_report(stats, 'nginx_report.md')
    print('报告已生成: nginx_report.md')
```

## 📝 实战案例三：JSON 转 CSV 批量处理

### 需求描述

```
帮我处理一个文件夹里的 JSON 文件：

- 文件夹：data/
- 每个 JSON 结构一样，但字段很多
- 只需要提取几个特定字段
- 合并成一个 CSV 文件
```

### Claude Code 生成的脚本

```python
import json
import csv
import os
from pathlib import Path

def json_to_csv(input_dir, output_csv, fields):
    """
    批量转换 JSON 到 CSV
    
    Args:
        input_dir: JSON 文件目录
        output_csv: 输出 CSV 路径
        fields: 要提取的字段列表，支持嵌套如 'user.name'
    """
    rows = []
    
    for json_file in Path(input_dir).glob('*.json'):
        with open(json_file, 'r', encoding='utf-8') as f:
            data = json.load(f)
        
        row = {}
        for field in fields:
            # 支持嵌套字段
            value = data
            for key in field.split('.'):
                value = value.get(key, '') if isinstance(value, dict) else ''
            row[field] = value
        
        rows.append(row)
    
    # 写入 CSV
    with open(output_csv, 'w', newline='', encoding='utf-8') as f:
        writer = csv.DictWriter(f, fieldnames=fields)
        writer.writeheader()
        writer.writerows(rows)
    
    print(f'处理完成: {len(rows)} 条记录 → {output_csv}')

# 使用示例
if __name__ == '__main__':
    fields = ['id', 'name', 'email', 'address.city', 'created_at']
    json_to_csv('data/', 'output.csv', fields)
```
