# 5.1 完整 Web 项目开发

> 从零到一，用 Claude Code 撸一个能跑的 Web 应用。

## 🎯 场景描述

你想快速做一个 Web 项目：用户登录、数据增删改查、页面展示。传统方式要写一堆文件，Claude Code 可以帮你一次性搞定。

## 💡 核心思路

```
一个完整需求 → Claude Code 生成骨架 → 补充细节 → 运行调试
```

## 📝 实战案例：Todo 应用

### 第一步：给 Claude Code 一个清晰的需求

```
帮我创建一个 Todo List Web 应用，要求：
- 后端用 Node.js + Express
- 前端用原生 HTML/CSS/JS（不用框架）
- 数据存储用 SQLite
```

**server.js**
```javascript
const express = require('express');
const Database = require('./database');

const app = express();
const db = new Database();

app.use(express.json());
app.use(express.static('public'));

// 获取所有任务
app.get('/api/todos', (req, res) => {
  res.json(db.getAll());
});

// 添加任务
app.post('/api/todos', (req, res) => {
  const { text } = req.body;
  const todo = db.add(text);
  res.json(todo);
});

// 更新任务状态
app.put('/api/todos/:id', (req, res) => {
  const { id } = req.params;
  const { completed } = req.body;
  db.update(id, completed);
  res.json({ success: true });
});

// 删除任务
app.delete('/api/todos/:id', (req, res) => {
  const { id } = req.params;
  db.delete(id);
  res.json({ success: true });
});

app.listen(3000, () => {
  console.log('Todo app running at http://localhost:3000');
});
```

**database.js**
```javascript
const Database = require('better-sqlite3');
const db = new Database('todos.db');

// 初始化表
db.exec(`
  CREATE TABLE IF NOT EXISTS todos (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    text TEXT NOT NULL,
    completed INTEGER DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
  )
`);

class TodoDB {
  getAll() {
    return db.prepare('SELECT * FROM todos ORDER BY created_at DESC').all();
  }

  add(text) {
    const result = db.prepare('INSERT INTO todos (text) VALUES (?)').run(text);
    return { id: result.lastInsertRowid, text, completed: 0 };
  }

  update(id, completed) {
    db.prepare('UPDATE todos SET completed = ? WHERE id = ?').run(completed ? 1 : 0, id);
  }

  delete(id) {
    db.prepare('DELETE FROM todos WHERE id = ?').run(id);
  }
}

module.exports = TodoDB;
```

**public/app.js**
```javascript
let todos = [];

// 加载任务列表
async function loadTodos() {
  const res = await fetch('/api/todos');
  todos = await res.json();
  renderTodos();
}

// 渲染任务列表
function renderTodos() {
  const list = document.getElementById('todoList');
  list.innerHTML = todos.map(todo => `
    <li class="${todo.completed ? 'completed' : ''}">
      <input type="checkbox" ${todo.completed ? 'checked' : ''} 
             onchange="toggleTodo(${todo.id}, this.checked)">
      <span>${escapeHtml(todo.text)}</span>
      <button class="delete-btn" onclick="deleteTodo(${todo.id})">删除</button>
    </li>
  `).join('');
}

// 添加任务
async function addTodo() {
  const input = document.getElementById('todoInput');
  const text = input.value.trim();
  
  if (!text) return;
  
  const res = await fetch('/api/todos', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ text })
  });
  
  const todo = await res.json();
  todos.unshift(todo);
  renderTodos();
  input.value = '';
}

// 切换完成状态
async function toggleTodo(id, completed) {
  await fetch(`/api/todos/${id}`, {
    method: 'PUT',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ completed })
  });
  
  const todo = todos.find(t => t.id === id);
  if (todo) todo.completed = completed;
  renderTodos();
}

// 删除任务
async function deleteTodo(id) {
  await fetch(`/api/todos/${id}`, { method: 'DELETE' });
  todos = todos.filter(t => t.id !== id);
  renderTodos();
}

// HTML 转义
function escapeHtml(text) {
  const div = document.createElement('div');
  div.textContent = text;
  return div.innerHTML;
}

// 页面加载时获取任务
loadTodos();
```
